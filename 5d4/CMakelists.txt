cmake_minimum_required(VERSION 3.1.0)

project(samp5_4 VERSION 1.0.0 LANGUAGES CXX)

set(CMAKE_CXX_STANDARD 14)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

set(CMAKE_AUTOMOC ON)
set(CMAKE_AUTORCC ON)
set(CMAKE_AUTOUIC ON)

if(CMAKE_VERSION VERSION_LESS "3.7.0")
    set(CMAKE_INCLUDE_CURRENT_DIR ON)
endif()

find_package(Qt6 COMPONENTS Widgets Core5Compat REQUIRED)

add_executable(samp5_4
    mainwindow.ui
    mainwindow.cpp
    qwcomboboxdelegate.cpp
    qwfloatspindelegate.cpp
    qwintspindelegate.cpp
    main.cpp
    res.qrc
)

target_link_libraries(samp5_4 Qt6::Widgets Qt6::Core5Compat)
